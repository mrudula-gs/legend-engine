package org.finos.legend.engine.plan.execution.stores.deephaven.test.shared;

import org.finos.legend.engine.plan.execution.stores.deephaven.test.DeephavenTestContainer;
import org.finos.legend.pure.generated.Root_meta_pure_functions_io_http_URL;
import org.finos.legend.pure.generated.Root_meta_pure_functions_io_http_URL_Impl;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class DeephavenCommands {
    private static final Logger LOGGER = LoggerFactory.getLogger(DeephavenCommands.class);
    private static final int DEEPHAVEN_PORT = 10000;

    public static final String START_SERVER_FUNCTION = "startDeephaven_String_1__URL_1_";
    public static final String STOP_SERVER_FUNCTION = "stopDeephaven_String_1__Boolean_1_";
    public static final String LOAD_STATIC_TABLE_FUNCTION = "loadStaticTable_String_1__String_1__String_1__Boolean_1_";
    public static final String GET_PSK_FUNCTION = "getPsk_String_1__String_1_";

    public static Root_meta_pure_functions_io_http_URL startServer(String imageTag) {
        LOGGER.debug("Starting Deephaven with imageTag: {}", imageTag);

        if (!DeephavenTestContainer.startDeephaven()) {
            throw new RuntimeException("Failed to start Deephaven container");
        }

        Root_meta_pure_functions_io_http_URL_Impl url = new Root_meta_pure_functions_io_http_URL_Impl("deephavenUrl");
        url._host(DeephavenTestContainer.deephavenContainer.getHost());
        url._port(DeephavenTestContainer.deephavenContainer.getMappedPort(DEEPHAVEN_PORT));
        url._path("/");

        return url;
    }

    public static void stopServer(String imageTag) {
        LOGGER.debug("Stopping Deephaven with imageTag: {}", imageTag);
        DeephavenTestContainer.stopDeephaven();
    }

    public static boolean loadStaticTable(String imageTag, String tableName, String jsonData) {
        LOGGER.debug("Loading static table: {} with data: {}", tableName, jsonData);
        if (DeephavenTestContainer.deephavenContainer == null || !DeephavenTestContainer.deephavenContainer.isRunning()) {
            throw new IllegalStateException("Deephaven container not running for imageTag: " + imageTag);
        }
        return DeephavenTestContainer.loadStaticTable(tableName, jsonData);
    }

    public static String getPsk(String imageTag) {
        LOGGER.debug("Getting PSK for imageTag: {}", imageTag);
        String psk = DeephavenTestContainer.getPsk();
        if (psk == null) {
            throw new IllegalStateException("PSK not available for imageTag: " + imageTag);
        }
        return psk;
    }
}
