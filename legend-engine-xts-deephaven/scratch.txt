    public static boolean startDeephaven(String imageTag)
    {
        if (deephavenContainer != null && deephavenContainer.isRunning())
        {
            System.out.println("Deephaven container already running; stopping it first.");
            stopDeephaven();
        }
        try
        {
            DockerImageName imageName = DockerImageName.parse(DEEPHAVEN_IMAGE);
            deephavenContainer = new GenericContainer<>(imageName)
                    .withExposedPorts(DEEPHAVEN_PORT)
                    .withPrivilegedMode(true)
                    .withCopyFileToContainer(
                            MountableFile.forClasspathResource("load_data.groovy"),
                            "/data/load_data.groovy"
                    )
                    .withEnv("START_OPTS", "-Xmx4g -Dauthentication.psk=" + STATIC_PSK + " -Ddeephaven.console.type=groovy -Ddeephaven.console.startupScript=/data/load_data.groovy");


            deephavenContainer.start();
            System.out.println("Started Deephaven container on host: " + deephavenContainer.getHost() + " port: " + deephavenContainer.getMappedPort(DEEPHAVEN_PORT));
            System.out.println("PSK set to: " + STATIC_PSK);
            System.out.println("Startup script set to: /data/load_data.groovy");

            // Wait longer for server and script to initialize
            Thread.sleep(20000);
            System.out.println("Container logs after startup: " + deephavenContainer.getLogs());
            return true;

//            deephavenContainer.start();
//            System.out.println("Started Deephaven container on host: " + deephavenContainer.getHost() +
//                    " port: " + deephavenContainer.getMappedPort(DEEPHAVEN_PORT));
//            System.out.println("PSK set to: " + STATIC_PSK);
//            System.out.println("Startup script set to: /data/load_data.py");
//
//            Thread.sleep(5000); // Wait for server and script to initialize
//
//
//            return true;
        }
        catch (Exception e)
        {
            System.err.println("Failed to start Deephaven container: " + e.getMessage());
            e.printStackTrace();
            stopDeephaven();
            return false;
        }
    }

    public static boolean stopDeephaven()
    {
        if (deephavenContainer != null && deephavenContainer.isRunning())
        {
            try
            {
                deephavenContainer.stop();
                System.out.println("Deephaven container stopped");
                return true;
            }
            catch (Exception e)
            {
                System.err.println("Failed to stop Deephaven container: " + e.getMessage());
                return false;
            }
            finally
            {
                deephavenContainer = null;
                discoveredPsk.set(STATIC_PSK);
            }
        }
        else
        {
            System.out.println("No Deephaven container running to stop");
            return true;
        }
    }

    public static boolean loadStaticTable(String imageTag, String tableName, String jsonData)
    {
        if (deephavenContainer == null || !deephavenContainer.isRunning())
        {
            System.err.println("Deephaven container not running for imageTag: " + imageTag);
            return false;
        }
        System.out.println("Data loading handled at startup via load_data.groovy for table '" + tableName + "'");
        return true;
    }
