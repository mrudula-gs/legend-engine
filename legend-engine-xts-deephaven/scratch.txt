# load_data.py
from deephaven import DynamicTableWriter
import deephaven.dtypes as dht

# Create table writer with column definitions
table_writer = DynamicTableWriter({"DesktopName": dht.string, "CityName": dht.string})
nds_desktops = table_writer.table

# Static test data
data = [
    {"DesktopName": "Alpha", "CityName": "Dallas"},
    {"DesktopName": "Beta", "CityName": "New York"}
]

# Populate the table
for row in data:
    table_writer.write_row(row["DesktopName"], row["CityName"])

print("Loaded nds_desktops with test data")
