public class LakehousePropertyMapping extends PropertyMapping
{
    public IngestRelationColumn relationalColumn;

    @Override
    public <T> T accept(PropertyMappingVisitor<T> visitor)
    {
        return visitor.visit(this);
    }
}


public class IngestRelationColumn
{
    public IngestRelationAccessor ingestAccessor;
    public String Column;
    public SourceInformation sourceInformation;
}



public class IngestRelationAccessor extends RelationElementAccessor
{
    public SourceInformation sourceInformation;

    public List<String> path;

    public Boolean metadata;
}



public class RelationElementAccessor implements Relation {
    private Object sourceElement;

    public RelationElementAccessor() {
    }

    public Object getSourceElement() {
        return this.sourceElement;
    }

    public RelationElementAccessor setSourceElement(Object sourceElement) {
        this.sourceElement = sourceElement;
        return this;
    }
}


public interface Relation {
}
