package org.finos.legend.engine.plan.execution.stores.deephaven.test;

import io.deephaven.client.impl.BarrageSession;
import io.deephaven.client.impl.BarrageSessionFactoryConfig;
import io.deephaven.client.impl.ClientConfig;
import io.deephaven.client.impl.SessionConfig;
import io.deephaven.qst.table.TicketTable;
import io.deephaven.uri.DeephavenTarget;
import org.apache.arrow.flight.FlightStream;
import org.apache.arrow.memory.RootAllocator;
import org.testcontainers.containers.GenericContainer;
import org.testcontainers.utility.DockerImageName;
import org.testcontainers.utility.MountableFile;

import java.util.concurrent.Executors;
import java.util.concurrent.ScheduledExecutorService;
import java.util.concurrent.atomic.AtomicReference;

public class DeephavenTestContainer {
    public static GenericContainer<?> deephavenContainer;
    private static final int DEEPHAVEN_PORT = 10000;
    private static final String STATIC_PSK = "myStaticPsk";
    private static final AtomicReference<String> discoveredPsk = new AtomicReference<>(STATIC_PSK);

    public static boolean startDeephaven(String imageTag) {
        if (deephavenContainer != null && deephavenContainer.isRunning()) {
            System.out.println("Deephaven container already running; stopping it first.");
            stopDeephaven();
        }
        try {
            DockerImageName imageName = DockerImageName.parse(imageTag);
            deephavenContainer = new GenericContainer<>(imageName)
                    .withExposedPorts(DEEPHAVEN_PORT)
                    .withPrivilegedMode(true)
                    .withCopyFileToContainer(
                        MountableFile.forClasspathResource("load_data.py"),
                        "/data/load_data.py"
                    )
                    .withEnv("START_OPTS", "-Xmx4g -Dauthentication.psk=" + STATIC_PSK);
            deephavenContainer.start();
            System.out.println("Started Deephaven container on host: " + deephavenContainer.getHost() + 
                " port: " + deephavenContainer.getMappedPort(DEEPHAVEN_PORT));
            System.out.println("PSK set to: " + STATIC_PSK);

            // Wait for server to start
            Thread.sleep(5000);

            // Verify file exists in /data
            System.out.println("Checking if /data/load_data.py exists...");
            org.testcontainers.containers.Container.ExecResult lsResult = deephavenContainer.execInContainer("ls", "/data/load_data.py");
            System.out.println("ls stdout: " + lsResult.getStdout());
            if (!lsResult.getStderr().isEmpty()) {
                System.err.println("ls stderr: " + lsResult.getStderr());
            }

            // Execute script using server's Python
            System.out.println("Executing load_data.py with server's Python...");
            org.testcontainers.containers.Container.ExecResult pythonResult = deephavenContainer.execInContainer(
                "/opt/deephaven/venv/bin/python", "/data/load_data.py"
            );
            System.out.println("Script stdout: " + pythonResult.getStdout());
            if (!pythonResult.getStderr().isEmpty()) {
                System.err.println("Script stderr: " + pythonResult.getStderr());
            }
            if (pythonResult.getExitCode() != 0) {
                System.err.println("Script failed with exit code: " + pythonResult.getExitCode());
                return false;
            }
            return true;
        } catch (Exception e) {
            System.err.println("Failed to start Deephaven container or run script: " + e.getMessage());
            e.printStackTrace();
            stopDeephaven();
            return false;
        }
    }

    public static boolean stopDeephaven() {
        if (deephavenContainer != null && deephavenContainer.isRunning()) {
            try {
                deephavenContainer.stop();
                System.out.println("Deephaven container stopped");
                return true;
            } catch (Exception e) {
                System.err.println("Failed to stop Deephaven container: " + e.getMessage());
                return false;
            } finally {
                deephavenContainer = null;
                discoveredPsk.set(STATIC_PSK);
            }
        } else {
            System.out.println("No Deephaven container running to stop");
            return true;
        }
    }

    public static boolean loadStaticTable(String imageTag, String tableName, String jsonData) {
        if (deephavenContainer == null || !deephavenContainer.isRunning()) {
            System.err.println("Deephaven container not running for imageTag: " + imageTag);
            return false;
        }
        System.out.println("Data loading handled via exec for table '" + tableName + "'");
        return true;
    }

    public static String getPsk() {
        return discoveredPsk.get();
    }

    public static void main(String[] args) {
        String imageTag = "dockerhub.site.com:8413/deephaven/server:latest";
        String tableName = "nds_desktops";
        String jsonData = "[{\"Desktop_ID\": 1, \"Desktop_Name\": \"Alpha\", \"City_Name\": \"Dallas\"}, {\"Desktop_ID\": 2, \"Desktop_Name\": \"Beta\", \"City_Name\": \"New York\"}]";

        if (startDeephaven(imageTag)) {
            loadStaticTable(imageTag, tableName, jsonData);

            try {
                ScheduledExecutorService scheduler = Executors.newScheduledThreadPool(4);
                DeephavenTarget target = DeephavenTarget.builder()
                        .host(deephavenContainer.getHost())
                        .port(deephavenContainer.getMappedPort(DEEPHAVEN_PORT))
                        .isSecure(false)
                        .build();
                ClientConfig clientConfig = ClientConfig.builder()
                        .target(target)
                        .build();
                SessionConfig sessionConfig = SessionConfig.builder()
                        .authenticationTypeAndValue("io.deephaven.authentication.psk.PskAuthenticationHandler " + STATIC_PSK)
                        .build();
                RootAllocator bufferAllocator = new RootAllocator();
                BarrageSessionFactoryConfig.Factory barrageSessionFactory = BarrageSessionFactoryConfig.builder()
                        .clientConfig(clientConfig)
                        .allocator(bufferAllocator)
                        .scheduler(scheduler)
                        .build()
                        .factory();

                try (BarrageSession session = barrageSessionFactory.newBarrageSession(sessionConfig)) {
                    System.out.println("Connected to Deephaven server at " + target.host() + ":" + target.port());

                    TicketTable ticketTable = TicketTable.fromQueryScopeField(tableName);
                    io.deephaven.client.impl.TableHandle tableHandle = session.session().execute(ticketTable);
                    try (FlightStream flightStream = session.stream(tableHandle.ticketId())) {
                        int rowCount = flightStream.getRoot().getRowCount();
                        System.out.println("Table 'nds_desktops' found with " + rowCount + " rows");
                        while (flightStream.next()) {
                            System.out.println("Received data batch from nds_desktops");
                        }
                    }
                    System.out.println("Successfully verified nds_desktops table");
                }
            } catch (Exception e) {
                System.err.println("Failed to verify nds_desktops table: " + e.getMessage());
                e.printStackTrace();
            }

            System.out.println("Access Deephaven at http://" + deephavenContainer.getHost() + ":" + 
                deephavenContainer.getMappedPort(DEEPHAVEN_PORT) + "/?psk=" + STATIC_PSK);
            try {
                Thread.sleep(60000);
            } catch (InterruptedException e) {
                System.err.println("Interrupted: " + e.getMessage());
            }
            stopDeephaven();
        } else {
            System.err.println("Failed to start Deephaven container");
        }
    }
}
