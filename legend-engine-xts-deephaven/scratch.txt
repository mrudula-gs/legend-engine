// TDS to CSV Conversion
function meta::pure::metamodel::relation::toCSV(tds:TDS<Any>[1], tableName:String[1]):String[1]
{
    let header = $tds->columns()->map(c | $c.name->toOne())->joinStrings(',');
    let rows = $tds->map(row | 
        $tds->columns()->map(col | 
            let val = $col->eval($row)->toOne();
            $val->match([
                d:Date[1] | format('%t{yyyy-MM-dd"T"HH:mm:ss.SSSSSSSSS}', $d),
                x:Any[1] | $x->toString()
            ]);
        )->joinStrings(',')
    )->joinStrings('\n');
    $tableName + '\n' + $header + '\n' + $rows;
}
