function meta::external::store::deephaven::executionPlan::platformBinding::legendJava::generateSortOperation(op: SortOperation[1], chain: Code[1], tableSpecClass: meta::external::language::java::metamodel::Type[1]): Code[1]
{
  // Determine sort type based on columns and directions
  let allAscending = $op.columns->all(c | $c.direction == meta::external::store::deephaven::metamodel::SortDirection.ASCENDING);
  let allDescending = $op.columns->all(c | $c.direction == meta::external::store::deephaven::metamodel::SortDirection.DESCENDING);
  
  if($op.columns->size() == 1 && $allAscending,
      // Case 1: Single ascending column
      | $chain->j_invoke('sort', [j_string($op.columns->at(0).name)], $tableSpecClass),
      | if($allAscending,
            // Case 1: Multi-column ascending
            | $chain->j_invoke('sort', $op.columns->map(c | j_string($c.name)), $tableSpecClass),
            | if($allDescending,
                  // Case 2: Multi-column descending
                  | $chain->j_invoke('sortDescending', $op.columns->map(c | j_string($c.name)), $tableSpecClass),
                  // Case 3: Mixed directions
                  | let sortColumns = $op.columns->map(c | 
                      if($c.direction == meta::external::store::deephaven::metamodel::SortDirection.ASCENDING,
                         | javaClass('io.deephaven.api.SortColumn')->j_invoke('asc', [javaClass('io.deephaven.api.ColumnName')->j_invoke('of', [j_string($c.name)])]),
                         | javaClass('io.deephaven.api.SortColumn')->j_invoke('desc', [javaClass('io.deephaven.api.ColumnName')->j_invoke('of', [j_string($c.name)])])
                      )
                    );
                    $chain->j_invoke('sort', [javaClass('java.util.Arrays')->j_invoke('asList', $sortColumns)], $tableSpecClass)
                )
          )
  );
}
