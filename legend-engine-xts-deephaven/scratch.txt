package org.finos.legend.engine.plan.execution.stores.deephaven.test.interpreted;

import org.eclipse.collections.impl.factory.Lists;
import org.eclipse.collections.impl.tuple.Tuples;
import org.finos.legend.engine.plan.execution.stores.deephaven.test.shared.DeephavenCommands;
import org.finos.legend.pure.runtime.java.interpreted.extension.BaseInterpretedExtension;
import org.finos.legend.pure.runtime.java.interpreted.extension.InterpretedExtension;

public class DeephavenTestInterpretedExtension extends BaseInterpretedExtension {
    public static final String START_SERVER_FUNCTION = "startDeephaven_String_1__URL_1_";
    public static final String STOP_SERVER_FUNCTION = "stopDeephaven_String_1__Boolean_1_";
    public static final String LOAD_STATIC_TABLE_FUNCTION = "loadStaticTable_String_1__String_1__String_1__Boolean_1_";
    public static final String GET_PSK_FUNCTION = "getPsk_String_1__String_1_";

    public DeephavenTestInterpretedExtension() {
        super(Lists.mutable.with(
            Tuples.pair(START_SERVER_FUNCTION, DeephavenStartCommand::new),
            Tuples.pair(STOP_SERVER_FUNCTION, DeephavenStopCommand::new),
            Tuples.pair(LOAD_STATIC_TABLE_FUNCTION, DeephavenLoadStaticTableCommand::new),
            Tuples.pair(GET_PSK_FUNCTION, DeephavenGetPskCommand::new)
        ));
    }

    public static InterpretedExtension extension() {
        return new DeephavenTestInterpretedExtension();
    }
}
