package org.finos.legend.engine.plan.execution.stores.deephaven.test.shared;

import org.eclipse.collections.api.factory.Maps;
import org.finos.legend.engine.plan.execution.stores.deephaven.test.DeephavenTestContainer;
import org.finos.legend.pure.generated.Root_meta_pure_functions_io_http_URL;
import org.finos.legend.pure.generated.Root_meta_pure_functions_io_http_URL_Impl;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.testcontainers.containers.GenericContainer;

import java.util.Map;
import java.util.Optional;

public class DeephavenCommands
{
    private static final Logger LOGGER = LoggerFactory.getLogger(DeephavenCommands.class);
    private static final int DEEPHAVEN_PORT = 10000;
    private static final Map<String, GenericContainer<?>> CONTAINERS = Maps.mutable.empty();

    /**
     * Starts a Deephaven container for the given image tag and returns its URL with PSK.
     */
    public static Root_meta_pure_functions_io_http_URL startDeephaven(String imageTag)
    {
        LOGGER.debug("Starting Deephaven docker image: {}", imageTag);

        GenericContainer<?> container = CONTAINERS.computeIfAbsent(imageTag, tag -> {
            if (!DeephavenTestContainer.startDeephaven()) {
                throw new RuntimeException("Failed to start Deephaven container for image: " + imageTag);
            }
            return DeephavenTestContainer.deephavenContainer; // Assumes public static field
        });

        String psk = DeephavenTestContainer.getPsk();
        if (psk == null) {
            throw new RuntimeException("PSK not available after starting Deephaven for image: " + imageTag);
        }

        Root_meta_pure_functions_io_http_URL_Impl url = new Root_meta_pure_functions_io_http_URL_Impl("deephavenUrl");
        url._host(container.getHost());
        url._port(container.getMappedPort(DEEPHAVEN_PORT));
        url._path("/?psk=" + psk);

        LOGGER.info("Deephaven started for {} at {}:{} with PSK: {}", imageTag, container.getHost(), container.getMappedPort(DEEPHAVEN_PORT), psk);
        return url;
    }

    /**
     * Stops the Deephaven container for the given image tag.
     */
    public static void stopDeephaven(String imageTag)
    {
        LOGGER.debug("Stopping Deephaven docker image: {}", imageTag);
        Optional.ofNullable(CONTAINERS.remove(imageTag)).ifPresent(container -> {
            if (DeephavenTestContainer.stopDeephaven()) {
                LOGGER.info("Deephaven stopped for image: {}", imageTag);
            } else {
                LOGGER.warn("Failed to stop Deephaven for image: {}", imageTag);
            }
        });
    }

    /**
     * Authenticates a PSK against the running Deephaven instance for the given image tag.
     */
    public static Boolean authenticate(String imageTag, String psk)
    {
        LOGGER.debug("Authenticating Deephaven for image: {}", imageTag);
        GenericContainer<?> container = CONTAINERS.get(imageTag);
        if (container == null || !container.isRunning()) {
            LOGGER.error("Deephaven container not running for image: {}", imageTag);
            return false;
        }
        Boolean result = DeephavenTestContainer.authenticate(psk);
        LOGGER.debug("Authentication result for PSK '{}': {}", psk, result);
        return result;
    }

    /**
     * Loads a static table into Deephaven for the given image tag.
     */
    public static Boolean loadStaticTable(String imageTag, String tableName, String jsonData)
    {
        LOGGER.debug("Loading static table '{}' for Deephaven image: {}", tableName, imageTag);
        GenericContainer<?> container = CONTAINERS.get(imageTag);
        if (container == null || !container.isRunning()) {
            LOGGER.error("Deephaven container not running for image: {}", imageTag);
            return false;
        }
        Boolean result = DeephavenTestContainer.loadStaticTable(tableName, jsonData);
        LOGGER.debug("Table load result for '{}': {}", tableName, result);
        return result;
    }

    /**
     * Retrieves the PSK for the Deephaven instance running with the given image tag.
     */
    public static String getPsk(String imageTag)
    {
        LOGGER.debug("Retrieving PSK for Deephaven image: {}", imageTag);
        GenericContainer<?> container = CONTAINERS.get(imageTag);
        if (container == null || !container.isRunning()) {
            LOGGER.error("Deephaven container not running for image: {}", imageTag);
            return null;
        }
        String psk = DeephavenTestContainer.getPsk();
        LOGGER.debug("PSK retrieved: {}", psk);
        return psk;
    }
}
