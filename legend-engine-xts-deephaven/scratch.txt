package org.finos.legend.engine.plan.execution.stores.deephaven.test;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.testcontainers.containers.GenericContainer;
import org.testcontainers.containers.output.Slf4jLogConsumer;
import org.testcontainers.utility.DockerImageName;

import java.util.concurrent.atomic.AtomicReference;

public class DeephavenTestContainer {
    private static final Logger LOGGER = LoggerFactory.getLogger(DeephavenTestContainer.class);
    private static GenericContainer<?> deephavenContainer;
    private static final String REGISTRY = "dockerhub.site.com:8413";
    private static final String DEEPHAVEN_IMAGE = REGISTRY + "/deephaven/server:latest";
    private static final int DEEPHAVEN_PORT = 10000;
    private static final AtomicReference<String> discoveredPsk = new AtomicReference<>(null);

    /**
     * Starts a Deephaven instance and extracts the PSK from logs.
     * @return true if started successfully, false otherwise
     */
    public static boolean startDeephaven() {
        if (deephavenContainer != null && deephavenContainer.isRunning()) {
            LOGGER.warn("Deephaven container already running; stopping it first.");
            stopDeephaven();
        }

        try {
            DockerImageName imageName = DockerImageName.parse(DEEPHAVEN_IMAGE);
            deephavenContainer = new GenericContainer<>(imageName)
                .withExposedPorts(DEEPHAVEN_PORT)
                .withPrivilegedMode(true) // Add --privileged to match working Podman command
                .withLogConsumer(new Slf4jLogConsumer(LOGGER).withPrefix("Deephaven"));

            // Log consumer to capture PSK
            deephavenContainer.withLogConsumer(frame -> {
                String log = frame.getUtf8StringWithoutLineEnding();
                LOGGER.debug("Container log: {}", log); // Log all lines for debugging
                if (log.contains("psk=")) {
                    String psk = log.substring(log.indexOf("psk=") + 4).trim();
                    if (psk.contains(" ")) {
                        psk = psk.substring(0, psk.indexOf(" "));
                    }
                    discoveredPsk.set(psk);
                    int mappedPort = deephavenContainer.getMappedPort(DEEPHAVEN_PORT);
                    LOGGER.info("Discovered PSK: {} | Connect URL: http://localhost:{}/?psk={}", 
                                psk, mappedPort, psk);
                }
            });

            long startTime = System.currentTimeMillis();
            LOGGER.info("Starting Deephaven container with image: {}", DEEPHAVEN_IMAGE);
            deephavenContainer.start();

            // Wait for PSK or timeout
            int waitSeconds = 120; // 2 minutes
            for (int i = 0; i < waitSeconds && discoveredPsk.get() == null; i++) {
                Thread.sleep(1000);
            }

            if (discoveredPsk.get() == null) {
                String logs = deephavenContainer.getLogs();
                LOGGER.error("PSK not found in logs after {} seconds. Full logs:\n{}", waitSeconds, logs);
                throw new RuntimeException("Failed to extract PSK from Deephaven logs");
            }

            long endTime = System.currentTimeMillis();
            LOGGER.info("Started Deephaven container on host: {} port: {} time taken (ms): {}", 
                        deephavenContainer.getHost(), 
                        deephavenContainer.getMappedPort(DEEPHAVEN_PORT), 
                        (endTime - startTime));
            return true;
        } catch (Exception e) {
            LOGGER.error("Failed to start Deephaven container", e);
            stopDeephaven();
            throw e; // Re-throw to see full stack trace
        }
    }

    /**
     * Stops the running Deephaven container.
     * @return true if stopped successfully or no container was running, false on error
     */
    public static boolean stopDeephaven() {
        if (deephavenContainer != null && deephavenContainer.isRunning()) {
            try {
                deephavenContainer.stop();
                LOGGER.info("Deephaven container stopped");
                return true;
            } catch (Exception e) {
                LOGGER.error("Failed to stop Deephaven container", e);
                return false;
            } finally {
                deephavenContainer = null;
                discoveredPsk.set(null);
            }
        } else {
            LOGGER.warn("No Deephaven container running to stop");
            return true;
        }
    }

    /**
     * Authenticates using the provided PSK against the discovered one.
     * @param psk Pre-Shared Key to validate
     * @return true if valid, false otherwise
     */
    public static boolean authenticate(String psk) {
        if (deephavenContainer == null || !deephavenContainer.isRunning()) {
            LOGGER.error("Deephaven container not running; cannot authenticate");
            return false;
        }
        String currentPsk = discoveredPsk.get();
        if (currentPsk == null) {
            LOGGER.error("No PSK discovered yet; cannot authenticate");
            return false;
        }
        boolean isValid = currentPsk.equals(psk);
        LOGGER.info("PSK authentication attempt with '{}': {}", psk, isValid ? "success" : "failure");
        return isValid;
    }

    /**
     * Loads a static table into Deephaven with the given JSON data.
     * @param tableName Name of the table to create
     * @param jsonData JSON string containing table data
     * @return true if loaded successfully, false on error
     */
    public static boolean loadStaticTable(String tableName, String jsonData) {
        if (deephavenContainer == null || !deephavenContainer.isRunning()) {
            LOGGER.error("Deephaven container not running; cannot load table");
            return false;
        }
        if (discoveredPsk.get() == null) {
            LOGGER.error("PSK not discovered; cannot load table");
            return false;
        }

        try {
            String host = deephavenContainer.getHost();
            int port = deephavenContainer.getMappedPort(DEEPHAVEN_PORT);
            String psk = discoveredPsk.get();
            LOGGER.info("Loading static table '{}' with data: {}", tableName, jsonData);
            System.out.println("Stub: Loaded table " + tableName + " with data: " + jsonData);
            return true;
        } catch (Exception e) {
            LOGGER.error("Failed to load static table '{}'", tableName, e);
            return false;
        }
    }

    /**
     * Gets the discovered PSK for use in tests.
     * @return The PSK extracted from logs, or null if not found
     */
    public static String getPsk() {
        return discoveredPsk.get();
    }

    // Native method implementations for Pure integration
    public static boolean startDeephavenNative() {
        return startDeephaven();
    }

    public static boolean stopDeephavenNative() {
        return stopDeephaven();
    }

    public static boolean authenticateNative(String psk) {
        return authenticate(psk);
    }

    public static boolean loadStaticTableNative(String tableName, String jsonData) {
        return loadStaticTable(tableName, jsonData);
    }

    public static void main(String[] args) {
        startDeephaven();
        System.out.println("PSK: " + getPsk());
        stopDeephaven();
    }
}
