import meta::external::store::deephaven::executionTest::utils::*;
import meta::external::store::deephaven::executionTest::TestStore;
import meta::external::store::deephaven::executionTest::TestRuntime;
import meta::pure::test::*;
import meta::pure::functions::string::*;

function meta::external::store::deephaven::executionTest::testQueryOnStore(): Boolean[1]
{
    let imageTag = 'dockerhub.site.com:8413/deephaven/server:latest';
    let url = startDeephaven($imageTag);
    
    // Pre-step: Load sample data (not tested directly)
    let sampleData = '[{"DesktopName": "Alpha", "CityName": "Dallas"}, {"DesktopName": "Beta", "CityName": "New York"}]';
    loadStaticTable($imageTag, 'nds_desktops', $sampleData);
    
    // Query: Select DesktopName where CityName = 'Dallas'
    let queryResult = #>{meta::external::store::deephaven::executionTest::TestStore.nds_desktops}#
        ->filter(t | $t.CityName == 'Dallas')
        ->select(t | $t.DesktopName)
        ->from(meta::external::store::deephaven::executionTest::TestRuntime)
        ->toJSONString();
    
    stopDeephaven($imageTag);
    
    // Assert result matches expected JSON
    let expected = '[{"DesktopName": "Alpha"}]';
    assertEquals($expected, $queryResult, 'Query result mismatch');
    true;
}

function meta::external::store::deephaven::executionTest::runTests(): Boolean[1]
{
    testQueryOnStore();
}
