// Adapter Definition
function <<PCT.adapter>> {PCT.adapterName='Deephaven'} meta::external::store::deephaven::executionTest::testAdapterForDeephavenExecution<X|o>(f:Function<{->X[o]}>[1]):X[o]
{
    meta::external::store::deephaven::executionTest::testAdapterForDeephavenExecutionImpl(
        $f,
        meta::external::store::deephaven::executionTest::utils::startDeephaven('latest')
    )
}

// Core Adapter Implementation
function meta::external::store::deephaven::executionTest::testAdapterForDeephavenExecutionImpl<X|o>(f:Function<{->X[o]}>[1], dbc:URL[1]):X[o]
{
    let debug = noDebug();
    let reprocessed = $f->meta::external::store::deephaven::executionTest::reprocess($dbc, $debug);
    let newFunction = $reprocessed.first;

    let extensions = meta::external::store::deephaven::executionPlan::platformBinding::legendJava::deephavenOnlyLegendJavaPlatformBindingExtensions();

    // Generate raw execution plan
    let rawPlan = meta::pure::executionPlan::executionPlan(
        $newFunction->cast(@FunctionDefinition<{->Any[*]}>),
        ^meta::pure::runtime::ExecutionContext(),
        $extensions,
        $debug
    );

    // Bind and transform plan
    let planBindToJava = $rawPlan->meta::pure::executionPlan::platformBinding::localBinding::bindTestPlanToPlatformLocallyWithClasspathExtensions('LegendJava');
    let plan = $planBindToJava->meta::protocols::pure::vX_X_X::transformation::fromPureGraph::executionPlan::transformPlan($extensions);
    let planAsJson = $plan->meta::json::toJSON(1000, meta::json::config(false, false, true, true));

    // Execute plan
    let resultJson = meta::legend::executePlanAsJSON($planAsJson, []);

    // Process result
    let funcReturnType = $f->functionReturnType().rawType->toOne();
    let json = $resultJson->meta::json::parseJSON();
    let res = $json->meta::json::fromJSON(
        $funcReturnType,
        ^meta::json::ExtendedJSONDeserializationConfig(typeKeyName='__TYPE', failOnUnknownProperties=true, nullReplacementInArray=TDSNull)
    );

    // Handle TDS result specifically
    if ($funcReturnType->instanceOf(Relation),
        | let tdsString = $res->cast(@TDS<Any>)->toString();
          $tdsString->stringToTDS(),
        | $res
    )->cast(@X)->toMultiplicity(@[o]);
}

// Reprocessing Logic
function meta::external::store::deephaven::executionTest::reprocess(dbc:URL[1], debug:DebugContext[1]):Pair<FunctionDefinition<Any>, ProcessingState>[1]
{
    | let expr = $inputFunction;
      let state = ^ProcessingState(
          database = ^meta::external::store::deephaven::metamodel::store::Store(name='DeephavenTestStore', package=meta::external::store::deephaven::executionTest),
          runtime = ^Runtime(
              connectionStores = ^ConnectionStore(
                  connection = $dbc,
                  element = meta::external::store::deephaven::metamodel::store::Store->getByFullPath('meta::external::store::deephaven::executionTest::DeephavenTestStore')
              )
          )
      );

      // Reprocess the expression
      let reprocessedExpr = $expr->evaluateAndDeactivate()->map(e | $e->reprocessExpr($state));
      let newFunction = ^FunctionDefinition<Any>(
          name = 'testFunction',
          package = test,
          expressionSequence = $reprocessedExpr->cast(@ValueSpecification)->toOneMany()
      );

      // Setup tables from TDS if present
      $reprocessedExpr->forEach(e | $e->setupTables($state));

      pair($newFunction->addWith($state.runtime->toOne()), $state);
}

// Expression Reprocessing
function meta::external::store::deephaven::executionTest::reprocessExpr(expr:Any[1], state:ProcessingState[1]):Any[1]
{
    $expr->match([
        fe:FunctionExpression[1] |
            let params = $fe.parametersValues->evaluateAndDeactivate()->map(p | $p->reprocessExpr($state));
            ^$fe(parametersValues = $params),
        iv:InstanceValue[1] |
            if ($iv.genericType.rawType->toOne()->instanceOf(TDS),
                | let tds = $iv.values->toOne()->cast(@TDS<Any>);
                  let tableName = 'table_' + random()->toString();
                  let storeAccessor = ^RelationStoreAccessor<Any>(
                      store = $state.database,
                      sourceElement = $tableName
                  );
                  $state->mutateAdd('csvs', $tds->toCSV($tableName));
                  $storeAccessor,
                | $iv
            ),
        lf:LambdaFunction<Any>[1] |
            let body = $lf.expressionSequence->evaluateAndDeactivate()->map(e | $e->reprocessExpr($state));
            ^$lf(expressionSequence = $body->cast(@ValueSpecification)->toOneMany()),
        ve:VariableExpression[1] | $ve,
        x:Any[1] | $x
    ])
}

// Add Runtime to Function
function meta::external::store::deephaven::executionTest::addWith(func:FunctionDefinition<Any>[1], runtime:Runtime[1]):FunctionDefinition<Any>[1]
{
    let withExpr = ^SimpleFunctionExpression(
        importGroup = system::imports::coreImport,
        func = from_T_m__Runtime_1__T_m_,
        functionName = 'with',
        genericType = $func->functionReturnType(),
        multiplicity = $func->functionReturnMultiplicity(),
        parametersValues = [
            ^InstanceValue(
                genericType = $func->genericType()->toOne(),
                multiplicity = PureOne,
                values = $func->expressionSequence
            ),
            ^InstanceValue(
                genericType = ^GenericType(rawType=Runtime),
                multiplicity = PureOne,
                values = $runtime
            )
        ]
    );
    ^$func(expressionSequence = $withExpr);
}

// Setup Tables from CSV
function meta::external::store::deephaven::executionTest::setupTables(expr:Any[1], state:ProcessingState[1]):Nil[0]
{
    if ($expr->instanceOf(RelationStoreAccessor),
        | let storeAccessor = $expr->cast(@RelationStoreAccessor<Any>);
          let tableName = $storeAccessor.sourceElement->toOne()->cast(@String);
          let csv = $state.csvs->filter(c | $c->contains($tableName))->toOne();
          meta::external::store::deephaven::executionTest::utils::createTableFromCSV($tableName, $csv, $state.runtime.connectionStores.connection->toOne()->cast(@URL)),
        | []
    );
}

// TDS to CSV Conversion
function meta::pure::metamodel::relation::toCSV(tds:TDS<Any>[1], tableName:String[1]):String[1]
{
    let header = $tds->columns()->map(c | $c.name->toOne())->joinStrings(',');
    let rows = $tds->map(row | 
        $tds->columns()->map(col | 
            let val = $col->eval($row)->toOne();
            $val->match([
                d:Date[1] | format('%t{yyyy-MM-dd"T"HH:mm:ss.SSSSSSSSS}', $d),
                x:Any[1] | $x->toString()
            ])
        )->joinStrings(',')
    )->joinStrings('\n');
    $tableName + '\n' + $header + '\n' + $rows;
}
