package _pure.plan.root;

import io.deephaven.client.impl.BarrageSession;
import io.deephaven.client.impl.TableHandle;
import io.deephaven.qst.table.TableSpec;
import io.deephaven.qst.table.TicketTable;
import java.math.*;
import java.util.*;
import java.util.function.*;
import java.util.stream.*;
import org.finos.legend.engine.plan.dependencies.domain.date.DayOfWeek;
import org.finos.legend.engine.plan.dependencies.domain.date.DurationUnit;
import org.finos.legend.engine.plan.dependencies.domain.date.PureDate;
import org.finos.legend.engine.plan.dependencies.util.Library;
import org.finos.legend.engine.plan.execution.stores.deephaven.specifics.IDeephavenExecutionNodeSpecifics;

public class Execute implements IDeephavenExecutionNodeSpecifics
{
    public TableHandle execute(BarrageSession session)
    {
        TableSpec ts = TicketTable.fromQueryScopeField("nds_desktops")
                                  .select("Desktop_ID", "City_Name");
        try
        {
            return session.session().execute(ts);
        }
        catch (io.deephaven.client.impl.TableHandle.TableHandleException e)
        {
            throw new RuntimeException(e);
        }
        catch (InterruptedException e)
        {
            throw new RuntimeException(e);
        }
    }
}
