private Result executeDeephavenNode(DeephavenExecutionNode node) {
    try {
        DeephavenSession deephavenSession = createDeephavenSession(node.connection);
        try (BarrageSession session = deephavenSession.getBarrageSession()) {
            LOGGER.info("Connected to Deephaven Server for node execution");
            IDeephavenExecutionNodeSpecifics specifics = compileAndInstantiateSpecifics(node);

            // Extract TableSpec line from source
            String classFullName = ((JavaPlatformImplementation) node.implementation).executionClassFullName;
            JavaClass executeClass = executionPlan.globalImplementationSupport.classes.stream()
                    .filter(cls -> (cls.package + "." + cls.name).equals(classFullName))
                    .findFirst()
                    .orElseThrow(() -> new EngineException("Execute class not found: " + classFullName));
            String sourceCode = executeClass.source;
            String tableSpecLine = extractTableSpecLine(sourceCode);

            TableHandle table = specifics.execute(session);
            return executeQueryAndStreamResults(session, table, node.connection.sourceSpec.url, tableSpecLine);
        }
    } catch (Exception e) {
        throw new EngineException("Error executing Deephaven query: " + e.getMessage(), e, ExceptionCategory.USER_EXECUTION_ERROR);
    }
}

private String extractTableSpecLine(String sourceCode) {
    // Simple regex to find TableSpec assignment (adjust as needed)
    Pattern pattern = Pattern.compile("TableSpec ts = (.*?);", Pattern.DOTALL);
    Matcher matcher = pattern.matcher(sourceCode);
    if (matcher.find()) {
        String tableSpec = matcher.group(1).trim();
        LOGGER.debug("Extracted TableSpec: {}", tableSpec);
        return tableSpec;
    }
    LOGGER.warn("Could not extract TableSpec from source: {}", sourceCode);
    return "unknown_query";
}

private Result executeQueryAndStreamResults(BarrageSession session, TableHandle table, URI serverUrl, String tableSpecLine) {
    FlightStream flightStream = session.stream(table.ticketId());
    Stream<Map<String, Object>> rowStream = StreamSupport.stream(new DeephavenRowSpliterator(flightStream), false)
            .onClose(() -> {
                try {
                    flightStream.close();
                } catch (Exception e) {
                    LOGGER.warn("Failed to close FlightStream", e);
                }
            });

    DeephavenExecutionActivity activity = new DeephavenExecutionActivity(serverUrl, tableSpecLine);
    List<org.finos.legend.engine.plan.execution.result.ExecutionActivity> activities = Collections.singletonList(activity);
    DeephavenStreamingResult deephavenResult = new DeephavenStreamingResult(activities);
    return new StreamingObjectResult<>(rowStream, deephavenResult.getResultBuilder(), deephavenResult);
}
