private Result executeDeephavenNode(DeephavenExecutionNode node)
{
    try
    {
        String specificsClassName = JavaHelper.getExecutionClassFullName((JavaPlatformImplementation) node.implementation);
        Class<?> specificsClass = ExecutionNodeJavaPlatformHelper.getClassToExecute(node, specificsClassName, executionState, identity);
        IDeephavenExecutionNodeSpecifics specifics = (IDeephavenExecutionNodeSpecifics) specificsClass.getConstructor().newInstance();
        DeephavenSession deephavenSession = createDeephavenSession(node.connection);

        try (BarrageSession session = deephavenSession.getBarrageSession())
        {
            System.out.println("Connected to Deephaven Server!");
            TableHandle table = specifics.execute(session);
            FlightStream flightStream = session.stream(table.ticketId());

            // Lazy stream of rows
            Stream<Map<String, Object>> rowStream = StreamSupport.stream(
                new Spliterator<Map<String, Object>>()
                {
                    private VectorSchemaRoot root = flightStream.getRoot();
                    private java.util.List<FieldVector> vectors = root.getFieldVectors();
                    private int rowIndex = 0;
                    private int rowCount = 0;
                    private boolean isClosed = false;

                    @Override
                    public boolean tryAdvance(java.util.function.Consumer<? super Map<String, Object>> action)
                    {
                        if (isClosed)
                        {
                            return false;
                        }
                        try
                        {
                            if (rowIndex >= rowCount && !flightStream.next())
                            {
                                closeStream();
                                return false;
                            }
                            if (rowIndex >= rowCount)
                            {
                                rowCount = root.getRowCount();
                                rowIndex = 0;
                            }
                            Map<String, Object> row = new HashMap<>(vectors.size());
                            for (FieldVector vector : vectors)
                            {
                                row.put(vector.getName(), vector.getObject(rowIndex));
                            }
                            action.accept(row);
                            rowIndex++;
                            return true;
                        }
                        catch (Exception e)
                        {
                            closeStream(); // Ensure cleanup on error
                            throw new RuntimeException("Error streaming Deephaven rows", e);
                        }
                    }

                    @Override
                    public Spliterator<Map<String, Object>> trySplit()
                    {
                        return null; // Non-parallel for simplicity
                    }

                    @Override
                    public long estimateSize()
                    {
                        return Long.MAX_VALUE; // Unknown size
                    }

                    @Override
                    public int characteristics()
                    {
                        return Spliterator.ORDERED;
                    }

                    private void closeStream()
                    {
                        if (!isClosed)
                        {
                            try
                            {
                                flightStream.close();
                            }
                            catch (Exception e)
                            {
                                // Log or ignore, but don’t throw checked exception
                                System.err.println("Failed to close FlightStream: " + e.getMessage());
                            }
                            isClosed = true;
                        }
                    }
                },
                false
            ).onClose(() -> {
                try
                {
                    flightStream.close();
                }
                catch (Exception e)
                {
                    // Wrap in unchecked exception or log
                    throw new RuntimeException("Failed to close FlightStream in onClose", e);
                }
            });

            return new StreamingObjectResult<>(rowStream, new StreamBuilder(), new DeephavenStreamingResult(Collections.emptyList()));
        }
        finally
        {
            deephavenSession.close();
        }
    }
    catch (Exception e)
    {
        throw new EngineException("Error executing Deephaven Query", e, ExceptionCategory.USER_EXECUTION_ERROR);
    }
}
