function meta::external::store::deephaven::executionPlan::generation::planExecution(
    sq: meta::pure::mapping::StoreQuery[1],
    ext: RoutedValueSpecification[0..1],
    m: Mapping[0..1],
    runtime: Runtime[0..1],
    exeCtx: meta::pure::runtime::ExecutionContext[1],
    extensions: Extension[*],
    debug: DebugContext[1]
): ExecutionNode[1]
{
    let dhCtx = if ($exeCtx->instanceOf(DeephavenExecutionContext),
                   | $exeCtx->cast(@DeephavenExecutionContext),
                   | ^DeephavenExecutionContext());
    let processOutput = meta::external::store::deephaven::pureToDeephaven::process($sq, $ext, $m, $runtime, $dhCtx, $extensions, $debug);
    let selectOp = $processOutput.dhQuery.operations->filter(o | $o->instanceOf(SelectOperation))
                                       ->map(o | $o->cast(@SelectOperation))
                                       ->toOne('No SelectOperation found in DeephavenQuery');
    let columns = $selectOp.columns->mapWithIndex((col, idx) | ^TDSColumn(
        name = $col.name,
        offset = $idx,
        type = meta::external::store::deephaven::pureToDeephaven::mapDeephavenTypeToPureType($col.type)
    ));
    
    ^DeephavenExecutionNode(
        connection = $runtime->toOne('Missing required runtime for Deephaven plan execution').connectionByElement($sq.store)->cast(@DeephavenConnection),
        query = $processOutput.dhQuery->toOne(),
        resultType = ^meta::pure::executionPlan::TDSResultType(
            type = TabularDataSet,
            tdsColumns = $columns
        )
    );
}


function meta::external::store::deephaven::pureToDeephaven::mapDeephavenTypeToPureType(dhType: meta::external::store::deephaven::metamodel::type::Type[1]): GenericType[1]
{
    $dhType->match([
        t: meta::external::store::deephaven::metamodel::type::BooleanType[1] | ^GenericType(rawType=Boolean),
        t: meta::external::store::deephaven::metamodel::type::ByteType[1]   | ^GenericType(rawType=Integer),
        t: meta::external::store::deephaven::metamodel::type::CharType[1]   | ^GenericType(rawType=String),
        t: meta::external::store::deephaven::metamodel::type::ShortType[1]  | ^GenericType(rawType=Integer),
        t: meta::external::store::deephaven::metamodel::type::IntType[1]    | ^GenericType(rawType=Integer),
        t: meta::external::store::deephaven::metamodel::type::LongType[1]   | ^GenericType(rawType=Integer),
        t: meta::external::store::deephaven::metamodel::type::FloatType[1]  | ^GenericType(rawType=Float),
        t: meta::external::store::deephaven::metamodel::type::DoubleType[1] | ^GenericType(rawType=Float),
        t: meta::external::store::deephaven::metamodel::type::StringType[1] | ^GenericType(rawType=String),
        t: meta::external::store::deephaven::metamodel::type::TimeType[1]   | ^GenericType(rawType=Date),
        t: meta::external::store::deephaven::metamodel::type::CustomType[1] | fail('CustomType not supported in TDSResultType mapping; specify a concrete DataType')
    ]);
}
